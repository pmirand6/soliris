}

function saveMed(parametros) {
  $.ajax({
    url: getCurrentHostname() + "/soliris/ajax/ajx.medico_form.php",
    type: "POST",
    data: parametros,
    success: function (opciones) {
      if (opciones.indexOf("ERROR") != 0) {
       
        if (getQuerystring("id")) {
          if (parametros.estado == "19") {
            Swal.fire({
              title: "Actualización Correcta. Desea actualizar la documentación?",
              icon: "info",
              showCancelButton: true,
              confirmButtonColor: "#3085d6",
              cancelButtonColor: "#d33",
              confirmButtonText: "Si",
              cancelButtonText: "No",
            }).then((result) => {
              if (result.value) {
                window.location.href =
                  getCurrentHostname() +
                  "/soliris/administrador/docs_medico.php?id=" +
                  getQuerystring("id") + '&oper=mod';
              } else {
                parent.location.reload();
              }
            });
          } else {
            Swal.fire({
                title: "Actualización Correcta",
                icon: "success",
                showCancelButton: true,
                confirmButtonText: "Cerrar",
                showCancelButton: false,
              }).then((result) =>{
                if(result.value){
                    parent.location.reload();
                }
              });
          }
        } else {
          window.location.href =
            getCurrentHostname() +
            "/soliris/administrador/docs_medico.php?id=" +
            opciones;
        }
      } else {
        alert(opciones);
      }
    },
  });
}

function getCurrentHostname() {
  var protocolo, url, var_port, port;

  protocolo = window.location.protocol;
  url = window.location.hostname;
  var_port = window.location.port;

  if (var_port !== "") {
    port = ":" + var_port;
  } else {
    port = "";
  }

  return protocolo + "//" + url + port;
}
